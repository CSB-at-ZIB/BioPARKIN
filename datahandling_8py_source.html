00027 # 
00028 #     Parses files of the PARKIN raw format (see below)
00029 #     into a dictionary.
00030 #     
00031 #     The entityMap is put into objects of type EntityData.
00032 #     
00033 #     
00034 #     Raw format::
00035 #         0 4
00036 #           2 4.85286 1
00037 #           4 7.61857 1
00038 #           17 40.8571 1
00039 #           18 0.411429 1
00040 #         1 4
00041 #           2 4.811 1
00042 #           4 6.972 1
00043 #           17 42.7 1
00044 #           18 0.331 1
00045 #      
00046 #      Time points: 0 and 1 have 4 entityMap points.
00047 #      
00048 #     
00049 #     @since: 2010-06-07
00050 #     
00051 def read_raw_data(filename, dataSet=None, parkinController=None):
00052     entityMap = OrderedDict()
00053 
00054     try:
00055         with open(filename, 'r') as f:
00056             for line in f:
00057                 parts = line.strip().split(" ")
00058                 if line[0] != " ":  # new time point
00059                     time = float(parts[0])
00060                 else: # new entry for current time point
00061                     dataID = parts[0]
00062                     value = float(parts[1])
00063                     # we disregard parts[2], i.e. the "1"
00064                     if dataID in entityMap:  # fetch existing entity
00065                         entity = entityMap[dataID]
00066                     else: # create new entry
00067                         #entry = ([],[])
00068                         entity = entitydata.EntityData()
00069                         if dataSet:
00070                             entity.setAssociatedDataSet(dataSet)
00071                         entityMap[dataID] = entity
00072                     entity.dataDescriptors.append(time)
00073                     entity.datapoints.append(value)
00074     except Exception, e:
00075         logging.error("Could not read file %s. Error: %s" % (filename, e))
00076         return
00077 
00078     for (id, entity) in entityMap.items():
00079         entity.originalId = id
00080         entity.originalFilename = filename
00081         if parkinController and parkinController.ActiveModelController and parkinController.ActiveModelController.sbmlModel:
00082             #entity.sbmlEntity = helpers.sbmlhelpers.findSbmlEntity(id, parkinController.ActiveModelController.sbmlModel)
00083             entity.sbmlEntity = findSbmlEntity(id, parkinController.ActiveModelController.sbmlModel)
00084 
00085     return entityMap

00090 #     Parses a CSV (comma separated values) file with experimental data.
00091 # 
00092 #     The CSV has to have a header line. Fields may be encapsulated by " (but they
00093 #     don't have to be). The value format is e.g. "3.14".
00094 # 
00095 #     Headers have the following convention:
00096 #         - First column called C{"Timepoint [%UNIT%]"}, e.g. C{"Timepoint [s]"}
00097 #         - Second column called C{"Unit"} for the time unit.
00098 #         - One column per Species ID, given the unit in square brackets. E.g: C{"LH [mg/ml]"}
00099 #         - Optionally: One column per Species Weight called "Weight" *immediately* after the ID column.
00100 #         - All other columns are handled as kind of "meta data". So, there can be e.g. a "patient id" column.
00101 #           The data of this column will be put in its own EntityData object with isMetaData = True.
00102 #

00242                 entityMap[columnID] = entityData

00317 #     Reads the results of a Fortran-BioParkin run.
00318 #     
00319 #     The list of SBML entities is used to associate columns in the entityData file
00320 #     with their corresponding entityData entities.
00321 #         
00322 #     format:
00323 #     C{[time [WHITESPACE] par 1 [WHITESPACE] par n}
00324 #     
00325 #     e.g.:
00326 #     C{0.0000000000000000E+00     1.4999999999999999E-02    0.0000000000000000E+00}
00327 #     
00328 #     
00329 def read_integrated_data(filename, listOfEntities, dataSet=None):
00330 
00331     try:
00332         csv.register_dialect("SimulationData", delimiter=' ', quotechar='"', skipinitialspace=True)
00333         reader = csv.reader(open(filename), dialect="SimulationData")
00334         time = []
00335         values = {}
00336         for (i, row) in enumerate(reader):  # we don't use the i... yet
00337 
00338             for (j, value) in enumerate(row):
00339                 if j == 0:  # jump over the 0th row
00340                     time.append(value) # time value is always in the 0th row
00341                     continue
00342 
00343                 columnID = j - 1 # possibility to compensate for column offset
00344                 entity = listOfEntities[columnID]
00345                 if values.has_key(entity):
00346                     values[entity].append(value) # append to existing list
00347                 else:
00348                     values[entity] = [value] # create list for the first time
00349 
00350         # create final data structure
00351         dataMap = {}
00352         if len(values) > 0: # if we have dataMap
00353             for entity in values.keys():
00354                 entityData = EntityData()
00355                 if dataSet:
00356                     entityData.setAssociatedDataSet(dataSet)
00357                 entityData.dataDescriptors = time
00358                 entityData.datapoints = values[entity]
00359                 entityData.type = entitydata.TYPE_SIMULATED
00360                 entityData.sbmlEntity = entity.wrappedEntity
00361                 entityData.originalFilename = filename
00362                 dataMap[entity] = entityData
00363             return dataMap
00364 
00365     except Exception, e:
00366         logging.error("Error while trying to read simulation dataMap file: %s" % e)
