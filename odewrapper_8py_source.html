00175             return False

00178             species = self.speciesEntity.Item
00179             if not type(species) == libsbml.Species:    # if the speciesEntity is no Species, something's wrong :)
00180             #if not self.speciesEntity == sbml_entities.Type[]
00181                 logging.debug("OdeWrapper.isValid(): self.speciesEntity holds no Species but a: %s" % type(species))
00182                 return False
00183 
00184             isConstant = species.getConstant()
00185             isBoundaryCondition = species.getBoundaryCondition()
00186             if isConstant and isBoundaryCondition: # Species quantity never changes
00187                 return False
00188             if not isConstant and isBoundaryCondition: # Species quantity only changed by Rules and Events
00189                 return False
00190             if isConstant and not isBoundaryCondition: # Species quantity never changes
00191                 return False
00192             if not isConstant and not isBoundaryCondition: # Species Quantity changed by Reactions *or* Rules, and Events
