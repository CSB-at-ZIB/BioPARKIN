00073     # 
00074     #         Takes the libSBML mathNode AST tree of this Reaction's
00075     #         KineticLaw and converts it into
00076     #         something that is valid FORTRAN code. (e.g. "pow(x,2)" becomes
00077     #         "x ** 2", etc.)
00078     #         
00079     #         
00080     def mathForFortran(self):
00081         if not self.kineticLaw:
00082             return 
00083         
00084         validFortran = helpers.handleMathNode(self.kineticLaw.getMath())
00085         
00086         # we start with replacing the longest IDs first. should help
00087         # to minimize problems with IDs that are substrings of longer IDs
00088         # Note: The problem still persists, it's not gone entirely. (Because
00089         # the replacement also contains the original ID and can still overlap
00090         # with other IDs.)
00091         sortedLocalParamIDs = sorted(self.localParameterIDs, key=len, reverse=True)
00092         
00093         for localParam in sortedLocalParamIDs:
00094             if localParam in validFortran:
00095                 validFortran = validFortran.replace(localParam, "%s_%s" % (self.id, localParam))
00096                 
00097         return validFortran

00125     #         This is duplicate code. It originaly appears in the parameterwrapper.But, here we have to handle a

00133     def getScope(self, param):
00134         if not type(param) == libsbml.Parameter:
00135             return
00136 
00137         parent = param.getParentSBMLObject() # parent= ListOfParameters
00138         if parent:
00139             grandpa = parent.getParentSBMLObject()
00140             if type(grandpa) == libsbml.Model:
00141                 return "Global"
00142             elif type(grandpa) == libsbml.KineticLaw:
00143                 reaction = grandpa.getParentSBMLObject()
00144                 return "%s" % reaction.getId()
00145 
00146     def getCombinedId(self, param):
00147         scope = self.getScope(param)
00148         if scope == "Global":
00149             return param.getId()
00150         else:
00151             return "%s_%s" % (scope, param.getId())
