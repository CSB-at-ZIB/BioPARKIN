00055     def getScope(self):
00056         if not type(self.wrappedParameter) == libsbml.Parameter:
00057             return
00058         
00059         parent = self.wrappedParameter.getParentSBMLObject() # parent= ListOfParameters
00060         if parent:
00061             grandpa = parent.getParentSBMLObject()
00062             if type(grandpa) == libsbml.Model:
00063                 return "Global"
00064             elif type(grandpa) == libsbml.KineticLaw:
00065                 reaction = grandpa.getParentSBMLObject()
00066                 return "%s" % reaction.getId()
00067
