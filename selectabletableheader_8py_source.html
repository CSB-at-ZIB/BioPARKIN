00105     def on_sectionClicked(self, logicalIndex):
00106         if logicalIndex in self._nonSelectableIndexes:
00107             return
00108             
00109         # swap the selection state
00110         if self._isSectionSelected[logicalIndex] == Qt.Checked:
00111             selected = Qt.Unchecked
00112         elif self._isSectionSelected[logicalIndex] == Qt.Unchecked:
00113             selected = Qt.Checked
00114         elif self._isSectionSelected[logicalIndex] == Qt.PartiallyChecked:
00115             selected = Qt.Checked # check partially checked cols/rows with the first click
00116             # Wikipedia says so! ;) http://en.wikipedia.org/wiki/Checkbox
00117             #  "The indeterminate state cannot usually be selected by the user, and switches to a checked state when activated."
00118         self._isSectionSelected[logicalIndex] = selected
00119 
00120         self.updateSection(logicalIndex)
00121         self.sectionSelectionChanged.emit(logicalIndex, selected)

00125     # 
00126     #         Whenever some selection in the supplied "selection model"
00127     #         is changed, we have to check whether this change affects
00128     #         one of self._isSectionSelected. That depends on whether
00129     #         self has vertical or horizontal orientation.
00130     # 
00131     #         We also have to check the state within the given "selection
00132     #         model" to assess whether the internal sections are completely
00133     #         checked/unchecked or in an intermediate state.
00134     #         (To be able to do this, self._isSectionSelected has to store "more"
00135     #         than just a bool. We now use a the Qt.Checkstate enum which has
00136     #         Qt.PartiallyChecked.)
00137     # 
00138     #         The indices given by the two QModelIndex correspond to the internal
00139     #         sections by design.
00140     #         
00141     def on_selectionModelChanged(self, topLeftIndex, bottomRightIndex):
00142         if self.orientation() == Qt.Horizontal: # default orientation
00143             affectedColLeft = topLeftIndex.column()
00144             affectedColRight = bottomRightIndex.column()
00145 
00146             # Now, loop over this area of the "selection model"
00147             for affectedSection in xrange(affectedColLeft, affectedColRight+1):
00148                 state = self._selectionModel.getColSelectionState(affectedSection)
00149                 if state is not None: # check against None explicitely, as it might very well be 0
00150                     self._isSectionSelected[affectedSection] = state
00151                 self.updateSection(affectedSection)
00152 
00153         elif self.orientation() == Qt.Vertical:
00154             affectedRowTop = topLeftIndex.row()
00155             affectedRowBottom = bottomRightIndex.row()
00156 
00157             # Now, loop over this area of the "selection model"
00158             for affectedSection in xrange(affectedRowTop, affectedRowBottom+1):
00159                 state = self._selectionModel.getRowSelectionState(affectedSection)
00160                 if state is not None: # check against None explicitely, as it might very well be 0
00161                     self._isSectionSelected[affectedSection] = state
00162                 self.updateSection(affectedSection)
