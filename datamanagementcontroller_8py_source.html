00125     def saveListItems(self):
00126         listOfFilenames = []
00127         for index in xrange(self.listWidget_Experimental.count()):
00128             listWidgetItem = self.listWidget_Experimental.item(index)
00129             item = listWidgetItem.data(Qt.DisplayRole) # use display role, because we want only the string
00130             listOfFilenames.append(item)
00131         
00132         self.DataService.set_experimental_data_files(listOfFilenames)
