00003 # A Python Singleton mixin class that makes use of some of the ideas
00004 # found at http://c2.com/cgi/wiki?PythonSingleton. Just inherit
00005 # from it and you have a singleton. No code is required in
00006 # subclasses to create singleton behavior -- inheritance from 
00007 # Singleton is all that is needed.
00008 # 
00009 # Singleton creation is threadsafe.
00010 # 
00011 # USAGE:
00012 # 
00013 # Just inherit from Singleton. If you need a constructor, include
00014 # an __init__() method in your class as you usually would. However,
00015 # if your class is S, you instantiate the singleton using S.getInstance() 
00016 # instead of S(). Repeated calls to S.getInstance() return the 
00017 # originally-created instance.
00018 # 
00019 # For example:
00020 # 
00021 # class S(Singleton):
00022 # 
00023 #     def __init__(self, a, b=1):
00024 #         pass
00025 #         
00026 # S1 = S.getInstance(1, b=3)
00027 # 
00028 # 
00029 # Most of the time, that's all you need to know. However, there are some
00030 # other useful behaviors. Read on for a full description:
00031 # 
00032 # 1) Getting the singleton:
00033 # 
00034 #     S.getInstance() 
00035 #     
00036 # returns the instance of S. If none exists, it is created. 
00037 # 
00038 # 2) The usual idiom to construct an instance by calling the class, i.e.
00039 # 
00040 #     S()
00041 #     
00042 # is disabled for the sake of clarity. 
00043 # 
00044 # For one thing, the S() syntax means instantiation, but getInstance()
00045 # usually does not cause instantiation. So the S() syntax would
00046 # be misleading.
00047 # 
00048 # Because of that, if S() were allowed, a programmer who didn't 
00049 # happen to notice the inheritance from Singleton (or who
00050 # wasn't fully aware of what a Singleton pattern
00051 # does) might think he was creating a new instance, 
00052 # which could lead to very unexpected behavior.
00053 # 
00054 # So, overall, it is felt that it is better to make things clearer
00055 # by requiring the call of a class method that is defined in
00056 # Singleton. An attempt to instantiate via S() will result 
00057 # in a SingletonException being raised.
00058 # 
00059 # 3) Use __S.__init__() for instantiation processing,
00060 # since S.getInstance() runs S.__init__(), passing it the args it has received. 
00061 # 
00062 # If no data needs to be passed in at instantiation time, you don't need S.__init__().
00063 # 
00064 # 4) If S.__init__(.) requires parameters, include them ONLY in the
00065 # first call to S.getInstance(). If subsequent calls have arguments,
00066 # a SingletonException is raised by default.
00067 # 
00068 # If you find it more convenient for subsequent calls to be allowed to
00069 # have arguments, but for those argumentsto be ignored, just include 
00070 # 'ignoreSubsequent = True' in your class definition, i.e.:
00071 # 
00072 #   class S(Singleton):
00073 #   
00074 #       ignoreSubsequent = True
00075 # 
00076 #       def __init__(self, a, b=1):
00077 #           pass
00078 # 
00079 # 5) For testing, it is sometimes convenient for all existing singleton
00080 # instances to be forgotten, so that new instantiations can occur. For that
00081 # reason, a forgetAllSingletons() function is included. Just call
00082 # 
00083 #   forgetAllSingletons()
00084 #   
00085 # and it is as if no earlier instantiations have occurred.
00086 # 
00087 # 6) As an implementation detail, classes that inherit 
00088 # from Singleton may not have their own __new__
00089 # methods. To make sure this requirement is followed, 
00090 # an exception is raised if a Singleton subclass includ
00091 # es __new__. This happens at subclass instantiation
00092 # time (by means of the MetaSingleton metaclass.
00093 #

00097 # how much it owes to other people's code and ideas which are in the
00098 # public domain. The idea of using a metaclass came from 
00099 # a  comment on Gary's blog (see 
00100 # http://www.garyrobinson.net/2004/03/python_singleto.html#comments). 
00101 # Other improvements came from comments and email from other
00102 # people who saw it online. (See the blog post and comments
00103 # for further credits.)

00115 _lockForSingletons = threading.RLock()
00116 _lockForSingletonCreation = threading.RLock()   # Ensure only one instance of each Singleton
00117                                                 # class is created.  This is not bound to the 
00118                                                 # individual Singleton class since we need to
00119                                                 # ensure that there is only one mutex for each
00120                                                 # Singleton class, which would require having
00121                                                 # a lock when setting up the Singleton class,
00122                                                 # which is what this is anyway.  So, when any
00123                                                 # Singleton is created, we lock this lock and
00124                                                 # then we don't need to lock it again for that
00125                                                 # class.

00127 def _createSingletonInstance(cls, lstArgs, dctKwArgs):
00128     _lockForSingletonCreation.acquire()
00129     try:
00130         if cls._isInstantiated(): # some other thread got here first
00131             return 
00132         
00133         instance = cls.__new__(cls)
00134         try:
00135             instance.__init__(*lstArgs, **dctKwArgs)
00136         except TypeError, e:
00137             if e.message.find('__init__() takes') != -1:
00138                 raise SingletonException, 'If the singleton requires __init__ args, supply them on first call to getInstance().' 
00139             else:
00140                 raise
00141         cls.cInstance = instance
00142         _addSingleton(cls)
00143     finally:
00144         _lockForSingletonCreation.release()

00146 def _addSingleton(cls):
00147     _lockForSingletons.acquire()
00148     try:
00149         assert cls not in _stSingletons
00150         _stSingletons.add(cls)
00151     finally:
00152         _lockForSingletons.release()

00154 def _removeSingleton(cls):
00155     _lockForSingletons.acquire()
00156     try:
00157         if cls in _stSingletons:
00158             _stSingletons.remove(cls)
00159     finally:
00160         _lockForSingletons.release()

00164 def forgetAllSingletons():
00165     _lockForSingletons.acquire()
00166     try:
00167         for cls in _stSingletons.copy():
00168             cls._forgetClassInstanceReferenceForTesting()
00169 
00170         # Might have created some Singletons in the process of tearing down.
00171         # Try one more time - there should be a limit to this.
00172         iNumSingletons = len(_stSingletons)
00173         if len(_stSingletons) > 0:
00174             for cls in _stSingletons.copy():
00175                 cls._forgetClassInstanceReferenceForTesting()
00176                 iNumSingletons -= 1
00177                 assert iNumSingletons == len(_stSingletons), 'Added a singleton while destroying ' + str(cls)
00178         assert len(_stSingletons) == 0, _stSingletons
00179     finally:
00180         _lockForSingletons.release()

00200     def getInstance(cls, *lstArgs, **dctKwArgs):
00201         if cls._isInstantiated():
00202             if (lstArgs or dctKwArgs) and not hasattr(cls, 'ignoreSubsequent'):
00203                 raise SingletonException, 'Singleton already instantiated, but getInstance() called with args.'
00204         else:
00205             _createSingletonInstance(cls, lstArgs, dctKwArgs)
00206             
00207         return cls.cInstance

00220     # 
00221     #         This is designed for convenience in testing -- sometimes you 
00222     #         want to get rid of a singleton during test code to see what
00223     #         happens when you call getInstance() under a new situation.
00224     #         
00225     #         To really delete the object, all external references to it
00226     #         also need to be deleted.
00227     #         
00228     def _forgetClassInstanceReferenceForTesting(cls):
00229         try:
00230             if hasattr(cls.cInstance, '_prepareToForgetSingleton'):
00231                 # tell instance to release anything it might be holding onto.
00232                 cls.cInstance._prepareToForgetSingleton()
00233             del cls.cInstance
00234             _removeSingleton(cls)
00235         except AttributeError:
00236             # run up the chain of base classes until we find the one that has the instance
00237             # and then delete it there
00238             for baseClass in cls.__bases__: 
00239                 if issubclass(baseClass, Singleton):
00240                     baseClass._forgetClassInstanceReferenceForTesting()
