00053     #         Recursive function to evaluate the type of the given libSBML

00059     def handle(self, node):
00060         if node is None:
00061             return None
00062 
00063         # handle node based on node type
00064         if node.getType() == libsbml.AST_INTEGER:
00065             return self.astConverterTemplate.handleInt(node)
00066         elif node.isReal():
00067             return self.astConverterTemplate.handleReal(node)
00068         elif node.getType() == libsbml.AST_NAME:
00069             if self.idsToReplace and node.getName() in self.idsToReplace.keys():
00070                 return self.idsToReplace[node.getName()]
00071             else:
00072                 return self.astConverterTemplate.handleString(node)
00073 
00074         elif node.getType() == libsbml.AST_PLUS:
00075             return self.astConverterTemplate.handlePlus(node)
00076         elif node.getType() == libsbml.AST_MINUS:
00077             return self.astConverterTemplate.handleMinus(node)
00078         elif node.getType() == libsbml.AST_DIVIDE:
00079             return self.astConverterTemplate.handleDivide(node)
00080         elif node.getType() == libsbml.AST_TIMES:
00081             return self.astConverterTemplate.handleTimes(node)
00082         elif node.getType() == libsbml.AST_POWER or node.getType() == libsbml.AST_FUNCTION_POWER:
00083             return self.astConverterTemplate.handlePower(node)
00084         elif node.getType() == libsbml.AST_FUNCTION_LN:
00085             return self.astConverterTemplate.handleLn(node)
00086         elif node.getType() == libsbml.AST_FUNCTION_LOG:
00087             return self.astConverterTemplate.handleLog(node)
00088         elif node.getType() == libsbml.AST_FUNCTION:
00089             # this has to be handled differently because in this case, we have
00090             # a function that is defined in the SBML itself
00091             if not self.mainModel:
00092                 logging.debug("AstConverter.handle(): Can't handle node of type AST_FUNCTION without a reference to the main SBML model.")
00093                 return None
00094             functionDefinition = self.mainModel.SbmlModel.Item.getFunctionDefinition(node.getName())
00095 
00096             numOfArguments = functionDefinition.getNumArguments()
00097             argsToReplace = {}
00098             for i in xrange(numOfArguments):
00099                 arg = functionDefinition.getArgument(i)
00100                 key = arg.getName()
00101                 valueNode = node.getChild(i)
00102                 value = self.handle(valueNode)
00103                 #TODO: elif valueNode.isFunction():
00104 
00105                 argsToReplace[key] = value
00106 
00107             mathNode = functionDefinition.getMath()
00108             return self.handle(mathNode)
00109         elif node.getType() == libsbml.AST_LAMBDA:
00110             # the right child contains the body of the math expression
00111             return self.handle(node.getRightChild())
00112

00114         logging.debug("AstConverter.handle(): Couldn't handle math node of type %s. The resulting expression will not be usable!" % node.getType())
